Example 1: Simple PRINT Statement
10 PRINT "HELLO, WORLD!"
20 END

Example 2: INPUT and PRINT Statements
10 INPUT A
20 PRINT A
20 END

Example 3: IF-THEN Statement
10 IF A > 10 THEN PRINT "A IS GREATER THAN 10"
20 END

Exemplo 4
10 POKE 100, 255
20 WAIT 100, 255
30 END

Exemplo 5
10 OPEN "MYFILE.TXT" FOR OUTPUT AS #1
20 PRINT #1, "HELLO, FILE!"
30 CLOSE #1
40 END

Example 6: GOTO Statement
10 PRINT "START"
20 GOTO 160
30 PRINT "THIS WON'T BE PRINTED"
40 GOTO 170
50 PRINT "GOTO WORKS"
60 PRINT "END"

Example 7: GOSUB and RETURN
10 GOSUB 50
20 PRINT "BACK FROM SUBROUTINE"
30 END
40 PRINT "IN SUBROUTINE"
50 RETURN


Example 8: Arithmetic Operations
10 LET X = 5
20 LET Y = 3
30 LET Z = X + Y * (X - Y) / X ^ Y
40 PRINT Z
50 END

Example 9
10 REM THIS IS A COMMENT
20 PRINT "COMMENTS DO NOT AFFECT EXECUTION"
30 END
